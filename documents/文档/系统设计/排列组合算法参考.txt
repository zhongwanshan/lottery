// 求组合
    utility.getCombination = function(n,m) {
        return utility.getFactorial(n)/(utility.getFactorial(m)*utility.getFactorial(n-m));
    };

    // 求阶乘
    utility.getFactorial = function (num) {
        if (num < 0) {
            return -1;
        } else if (num === 0 || num === 1) {
            return 1;
        } else {
            for (var i = num - 1; i >= 1; i--) {
                num *= i;
            }
        }
        return num;
    };
